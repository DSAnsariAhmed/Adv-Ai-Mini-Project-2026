# IMPORT LIBRARIES


import numpy as np   For numerical operations (arrays, matrices)
import matplotlib.pyplot as plt   For plotting images
import tensorflow as tf   Deep learning framework
from tensorflow.keras import layers, models, datasets   Keras modules
from sklearn.metrics.pairwise import cosine_similarity   For recommendation


# LOAD DATASET


 Load Fashion-MNIST dataset (images + labels)
(x_train, y_train), (x_test, y_test) = datasets.fashion_mnist.load_data()


# PREPROCESS DATA


def preprocess(imgs):
    imgs = imgs.astype("float32") / 255.0  
     Convert pixel values from 0–255 → 0–1 (normalization)

    imgs = np.expand_dims(imgs, -1)  
     Add channel dimension → (28,28) → (28,28,1)

    return imgs

 Apply preprocessing
x_train = preprocess(x_train)
x_test = preprocess(x_test)


# BUILD AUTOENCODER MODEL


latent_dim = 64  
 Size of compressed representation (feature vector)

 -------- ENCODER --------
encoder_input = layers.Input(shape=(28,28,1))  
 Input image shape

x = layers.Conv2D(32,3,activation='relu',padding='same')(encoder_input)  
 Apply convolution → extract features

x = layers.MaxPooling2D()(x)  
 Reduce image size (downsampling)

x = layers.Conv2D(64,3,activation='relu',padding='same')(x)  
 Extract deeper features

x = layers.MaxPooling2D()(x)  
 Further reduce size

x = layers.Flatten()(x)  
 Convert 2D feature map → 1D vector

latent = layers.Dense(latent_dim)(x)  
 Final compressed representation (latent vector)

 Create encoder model
encoder = models.Model(encoder_input, latent)


 -------- DECODER --------
decoder_input = layers.Input(shape=(latent_dim,))  
 Input is latent vector

x = layers.Dense(7*7*64)(decoder_input)  
 Expand latent vector

x = layers.Reshape((7,7,64))(x)  
 Convert to 3D feature map

x = layers.Conv2DTranspose(64,3,strides=2,activation='relu',padding='same')(x)  
 Upsample image

x = layers.Conv2DTranspose(32,3,strides=2,activation='relu',padding='same')(x)  
 Further upsample

decoder_output = layers.Conv2D(1,3,activation='sigmoid',padding='same')(x)  
 Final output image (grayscale)

 Create decoder model
decoder = models.Model(decoder_input, decoder_output)


 -------- AUTOENCODER --------
autoencoder = models.Model(encoder_input, decoder(encoder(encoder_input)))  
 Connect encoder + decoder

autoencoder.compile(optimizer='adam', loss='mse')  
 Compile model using MSE loss


# TRAIN MODEL


autoencoder.fit(
    x_train, x_train,  
     Input = Output (unsupervised learning)

    epochs=10,  
     Number of training iterations

    batch_size=128,  
     Number of samples per step

    validation_data=(x_test, x_test)  
     Evaluate on test data
)


 # IMAGE RECONSTRUCTION


encoded_imgs = encoder.predict(x_test[:10])  
 Convert images → latent vectors

decoded_imgs = decoder.predict(encoded_imgs)  
 Convert latent vectors → images

plt.figure(figsize=(10,4))

for i in range(10):
     Original images
    plt.subplot(2,10,i+1)
    plt.imshow(x_test[i].squeeze(), cmap='gray')
    plt.axis('off')

     Reconstructed images
    plt.subplot(2,10,i+11)
    plt.imshow(decoded_imgs[i].squeeze(), cmap='gray')
    plt.axis('off')

plt.show()


# IMAGE GENERATION

random_latent = np.random.normal(size=(10, latent_dim))  
 Generate random vectors

generated = decoder.predict(random_latent)  
 Convert to images

plt.figure(figsize=(10,2))

for i in range(10):
    plt.subplot(1,10,i+1)
    plt.imshow(generated[i].squeeze(), cmap='gray')
    plt.axis('off')

plt.show()


# RECOMMENDATION SYSTEM


train_features = encoder.predict(x_train[:1000])  
 Extract features from training images

test_img = x_test[0].reshape(1,28,28,1)  
 Select one test image

test_feature = encoder.predict(test_img)  
 Convert it to feature vector

similarities = cosine_similarity(test_feature, train_features)  
 Compute similarity scores

top_indices = np.argsort(similarities[0])[-5:]  
 Get indices of top 5 similar images

plt.figure(figsize=(10,2))

# Show query image
plt.subplot(1,6,1)
plt.imshow(x_test[0].squeeze(), cmap='gray')
plt.title("Query")
plt.axis('off')

# Show recommended images
for i, idx in enumerate(top_indices): <br>
   </t> plt.subplot(1,6,i+2)<br>
   </t> plt.imshow(x_train[idx].squeeze(), cmap='gray')<br>
   </t> plt.axis('off')<br>

plt.show()
